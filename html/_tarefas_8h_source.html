235double haversine(Place place_A, Place place_B) {
236
237 double lat_A = toRadians(place_A.get_latitude());
238 double lon_A= toRadians(place_A.get_longitude());
239 double lat_B = toRadians(place_B.get_latitude());
240 double lon_B = toRadians(place_B.get_longitude());
241
242
243 double dlat = lat_B - lat_A;
244 double dlon = lon_B - lon_A;
245 double a = std::sin(dlat / 2) * std::sin(dlat / 2) + std::cos(lat_A) * std::cos(lat_B) * std::sin(dlon / 2) * std::sin(dlon / 2);
246 double c = 2 * std::atan2(std::sqrt(a), std::sqrt(1 - a));
247 double distance = EARTH_RADIUS_KM * c;
248
249 return distance;
250}
